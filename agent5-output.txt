**TypeScript Specification:**
```typescript
interface HeaderComponentProps {
  logoUrl: string;
  navigationItems: NavigationItem[];
  userProfile: UserProfile;
  searchEnabled: boolean;
  onNavigationChange: (tabId: string) => void;
  onSearch: (query: string) => void;
  onUserAction: (action: string) => void;
}

interface HeaderComponentState {
  activeTab: string;
  searchQuery: string;
  isUserMenuOpen: boolean;
  isLoading: boolean;
}

class HeaderComponent extends React.Component<HeaderComponentProps, HeaderComponentState> {
  constructor(props: HeaderComponentProps) {
    super(props);
    this.state = {
      activeTab: 'dashboard',
      searchQuery: '',
      isUserMenuOpen: false,
      isLoading: false
    };
  }

  handleNavigationChange = (tabId: string) => {
    this.setState({ activeTab: tabId });
    this.props.onNavigationChange(tabId);
  };

  handleSearch = (query: string) => {
    this.setState({ searchQuery: query });
    this.props.onSearch(query);
  };

  handleUserMenuToggle = () => {
    this.setState(prevState => ({ isUserMenuOpen: !prevState.isUserMenuOpen }));
  };

  handleUserAction = (action: string) => {
    this.props.onUserAction(action);
    this.setState({ isUserMenuOpen: false });
  };

  render() {
    const { logoUrl, navigationItems, userProfile, searchEnabled } = this.props;
    const { activeTab, searchQuery, isUserMenuOpen, isLoading } = this.state;

    return (
      <aava-header
        logo-url={logoUrl}
        search-enabled={searchEnabled}
        user-profile={JSON.stringify(userProfile)}
        active-tab={activeTab}
        loading={isLoading}
      >
        <div slot="navigation">
          {navigationItems.map(item => (
            <aava-button
              key={item.id}
              variant={activeTab === item.id ? 'primary' : 'secondary'}
              onClick={() => this.handleNavigationChange(item.id)}
            >
              {item.label}
            </aava-button>
          ))}
        </div>
        
        {searchEnabled && (
          <div slot="search">
            <aava-textbox
              placeholder="Search..."
              value={searchQuery}
              onChange={(e) => this.handleSearch(e.target.value)}
            />
          </div>
        )}
        
        <div slot="user-menu">
          <aava-dropdown
            trigger-element={<aava-avatar src={userProfile.avatarUrl} />}
            open={isUserMenuOpen}
            onToggle={this.handleUserMenuToggle}
          >
            <aava-button onClick={() => this.handleUserAction('profile')}>Profile</aava-button>
            <aava-button onClick={() => this.handleUserAction('settings')}>Settings</aava-button>
            <aava-divider />
            <aava-button onClick={() => this.handleUserAction('logout')}>Logout</aava-button>
          </aava-dropdown>
        </div>
      </aava-header>
    );
  }
}
```

interface SidebarComponentProps {
  menuItems: MenuItem[];
  isCollapsed: boolean;
  userProfile: UserProfile;
  onMenuSelect: (itemId: string) => void;
  onCollapseToggle: (collapsed: boolean) => void;
}

interface SidebarComponentState {
  selectedMenuItem: string;
  collapseState: boolean;
  isLoading: boolean;
}

class SidebarComponent extends React.Component<SidebarComponentProps, SidebarComponentState> {
  constructor(props: SidebarComponentProps) {
    super(props);
    this.state = {
      selectedMenuItem: 'dashboard',
      collapseState: props.isCollapsed,
      isLoading: false
    };
  }

  handleMenuSelect = (itemId: string) => {
    this.setState({ selectedMenuItem: itemId });
    this.props.onMenuSelect(itemId);
  };

  handleCollapseToggle = () => {
    const newCollapseState = !this.state.collapseState;
    this.setState({ collapseState: newCollapseState });
    this.props.onCollapseToggle(newCollapseState);
  };

  render() {
    const { menuItems, userProfile } = this.props;
    const { selectedMenuItem, collapseState, isLoading } = this.state;

    return (
      <aava-sidebar
        collapsed={collapseState}
        loading={isLoading}
        user-profile={JSON.stringify(userProfile)}
      >
        <div slot="header">
          <aava-button
            variant="ghost"
            size="small"
            onClick={this.handleCollapseToggle}
          >
            <aava-icon name={collapseState ? 'menu-expand' : 'menu-collapse'} />
          </aava-button>
        </div>
        
        <div slot="navigation">
          <aava-menu selected-item={selectedMenuItem}>
            {menuItems.map(item => (
              <aava-menu-item
                key={item.id}
                id={item.id}
                icon={item.icon}
                disabled={item.disabled}
                onClick={() => this.handleMenuSelect(item.id)}
              >
                {!collapseState && item.label}
              </aava-menu-item>
            ))}
          </aava-menu>
        </div>
        
        <div slot="footer">
          {!collapseState && (
            <div className="user-info">
              <aava-avatar src={userProfile.avatarUrl} size="small" />
              <span className="user-name">{userProfile.name}</span>
            </div>
          )}
        </div>
      </aava-sidebar>
    );
  }
}
```

interface MetricCardProps {
  title: string;
  value: number | string;
  trend: 'up' | 'down' | 'neutral';
  icon: string;
  variant: 'primary' | 'secondary' | 'success' | 'warning';
  onCardClick?: () => void;
}

interface MetricCardState {
  isLoading: boolean;
  data: MetricData | null;
  error: string | null;
}

class MetricCard extends React.Component<MetricCardProps, MetricCardState> {
  constructor(props: MetricCardProps) {
    super(props);
    this.state = {
      isLoading: false,
      data: null,
      error: null
    };
  }

  handleCardClick = () => {
    if (this.props.onCardClick) {
      this.props.onCardClick();
    }
  };

  getTrendIcon = (trend: string) => {
    switch (trend) {
      case 'up': return 'arrow-up';
      case 'down': return 'arrow-down';
      default: return 'minus';
    }
  };

  getTrendColor = (trend: string) => {
    switch (trend) {
      case 'up': return 'success';
      case 'down': return 'danger';
      default: return 'neutral';
    }
  };

  render() {
    const { title, value, trend, icon, variant } = this.props;
    const { isLoading, error } = this.state;

    if (error) {
      return (
        <aava-default-card variant="error" clickable>
          <div className="error-state">
            <aava-icon name="alert-circle" />
            <span>Error loading metric</span>
          </div>
        </aava-default-card>
      );
    }

    return (
      <aava-default-card
        variant={variant}
        clickable={!!this.props.onCardClick}
        loading={isLoading}
        onClick={this.handleCardClick}
      >
        <div className="metric-card-content">
          <div className="metric-header">
            <aava-icon name={icon} size="large" />
            <h3 className="metric-title">{title}</h3>
          </div>
          
          <div className="metric-value">
            <span className="value">{value}</span>
            <div className={`trend trend-${trend}`}>
              <aava-icon 
                name={this.getTrendIcon(trend)} 
                color={this.getTrendColor(trend)}
                size="small"
              />
            </div>
          </div>
        </div>
      </aava-default-card>
    );
  }
}
```

```typescript
interface TimelineEvent {
  id: string;
  title: string;
  description: string;
  timestamp: Date;
  type: 'info' | 'success' | 'warning' | 'error';
  user?: string;
  metadata?: Record<string, any>;
}

interface ActivityTimelineProps {
  events: TimelineEvent[];
  orientation: 'vertical' | 'horizontal';
  size: 'small' | 'medium' | 'large';
  onEventSelect?: (event: TimelineEvent) => void;
}

interface ActivityTimelineState {
  selectedEvent: TimelineEvent | null;
  isLoading: boolean;
  error: string | null;
}

class ActivityTimeline extends React.Component<ActivityTimelineProps, ActivityTimelineState> {
  constructor(props: ActivityTimelineProps) {
    super(props);
    this.state = {
      selectedEvent: null,
      isLoading: false,
      error: null
    };
  }

  handleEventSelect = (event: TimelineEvent) => {
    this.setState({ selectedEvent: event });
    if (this.props.onEventSelect) {
      this.props.onEventSelect(event);
    }
  };

  formatTimestamp = (timestamp: Date) => {
    return new Intl.DateTimeFormat('en-US', {
      month: 'short',
      day: 'numeric',
      hour: '2-digit',
      minute: '2-digit'
    }).format(timestamp);
  };

  getEventIcon = (type: string) => {
    switch (type) {
      case 'success': return 'check-circle';
      case 'warning': return 'alert-triangle';
      case 'error': return 'alert-circle';
      default: return 'info';
    }
  };

  render() {
    const { events, orientation, size } = this.props;
    const { isLoading, error } = this.state;

    if (error) {
      return (
        <div className="timeline-error">
          <aava-icon name="alert-circle" />
          <span>Error loading timeline</span>
        </div>
      );
    }

    return (
      <aava-timeline
        orientation={orientation}
        size={size}
        loading={isLoading}
      >
        {events.map(event => (
          <aava-timeline-item
            key={event.id}
            type={event.type}
            timestamp={this.formatTimestamp(event.timestamp)}
            onClick={() => this.handleEventSelect(event)}
          >
            <div slot="icon">
              <aava-icon name={this.getEventIcon(event.type)} />
            </div>
            
            <div slot="content" className="timeline-content">
              <h4 className="event-title">{event.title}</h4>
              <p className="event-description">{event.description}</p>
              {event.user && (
                <div className="event-user">
                  <aava-avatar src={`/avatars/${event.user}`} size="small" />
                  <span>{event.user}</span>
                </div>
              )}
            </div>
          </aava-timeline-item>
        ))}
      </aava-timeline>
    );
  }
}
```

### ActionButton Specification

**Component Type:** Shared Component
**Purpose:** Trigger user actions and workflows using AAVA Play button component
**Parent:** Various (DashboardPage, DataTable, etc.)
**Children:** None (leaf component)

**TypeScript Specification:**
```typescript
interface ActionButtonProps {
  label: string;
  variant: 'primary' | 'secondary' | 'danger' | 'ghost';
  size: 'small' | 'medium' | 'large';
  disabled?: boolean;
  loading?: boolean;
  icon?: string;
  onClick: () => void;
}

interface ActionButtonState {
  isProcessing: boolean;
  error: string | null;
}

class ActionButton extends React.Component<ActionButtonProps, ActionButtonState> {
  constructor(props: ActionButtonProps) {
    super(props);
    this.state = {
      isProcessing: false,
      error: null
    };
  }

  handleClick = async () => {
    if (this.props.disabled || this.state.isProcessing) {
      return;
    }

    this.setState({ isProcessing: true, error: null });
    
    try {
      await this.props.onClick();
    } catch (error) {
      this.setState({ error: error.message });
    } finally {
      this.setState({ isProcessing: false });
    }
  };

  render() {
    const { label, variant, size, disabled, loading, icon } = this.props;
    const { isProcessing, error } = this.state;
    
    const isDisabled = disabled || isProcessing;
    const isLoading = loading || isProcessing;

    return (
      <div className="action-button-wrapper">
        <aava-button
          variant={variant}
          size={size}
          disabled={isDisabled}
          loading={isLoading}
          onClick={this.handleClick}
        >
          {icon && !isLoading && (
            <aava-icon name={icon} slot="icon" />
          )}
          {label}
        </aava-button>
        
        {error && (
          <div className="button-error">
            <aava-icon name="alert-circle" size="small" />
            <span>{error}</span>
          </div>
        )}
      </div>
    );
  }
}
```

### DataTable Specification

**Component Type:** Feature Component
**Purpose:** Display and manage structured data with sorting, filtering, and pagination
**Parent:** DataManagementPage
**Children:** FilterPanel, PaginationControls

**TypeScript Specification:**
```typescript
interface TableColumn {
  id: string;
  label: string;
  sortable: boolean;
  filterable: boolean;
  width?: string;
  align?: 'left' | 'center' | 'right';
}

interface TableRow {
  id: string;
  [key: string]: any;
}

interface FilterState {
  [columnId: string]: string | string[];
}

interface DataTableProps {
  columns: TableColumn[];
  data: TableRow[];
  sortable: boolean;
  filterable: boolean;
  paginated: boolean;
  pageSize?: number;
  onSort?: (column: string, direction: 'asc' | 'desc') => void;
  onFilter?: (filters: FilterState) => void;
  onPageChange?: (page: number) => void;
  onRowSelect?: (row: TableRow) => void;
}

interface DataTableState {
  sortColumn: string;
  sortDirection: 'asc' | 'desc';
  currentPage: number;
  filters: FilterState;
  selectedRows: string[];
  isLoading: boolean;
  error: string | null;
}

class DataTable extends React.Component<DataTableProps, DataTableState> {
  constructor(props: DataTableProps) {
    super(props);
    this.state = {
      sortColumn: '',
      sortDirection: 'asc',
      currentPage: 1,
      filters: {},
      selectedRows: [],
      isLoading: false,
      error: null
    };
  }

  handleSort = (column: string) => {
    const direction = this.state.sortColumn === column && this.state.sortDirection === 'asc' ? 'desc' : 'asc';
    this.setState({ sortColumn: column, sortDirection: direction });
    
    if (this.props.onSort) {
      this.props.onSort(column, direction);
    }
  };

  handleFilter = (columnId: string, value: string | string[]) => {
    const newFilters = { ...this.state.filters, [columnId]: value };
    this.setState({ filters: newFilters, currentPage: 1 });
    
    if (this.props.onFilter) {
      this.props.onFilter(newFilters);
    }
  };

  handlePageChange = (page: number) => {
    this.setState({ currentPage: page });
    
    if (this.props.onPageChange) {
      this.props.onPageChange(page);
    }
  };

  handleRowSelect = (rowId: string) => {
    const selectedRows = this.state.selectedRows.includes(rowId)
      ? this.state.selectedRows.filter(id => id !== rowId)
      : [...this.state.selectedRows, rowId];
    
    this.setState({ selectedRows });
  };

  renderTableHeader = () => {
    const { columns, sortable } = this.props;
    const { sortColumn, sortDirection } = this.state;

    return (
      <thead>
        <tr>
          <th>
            <aava-checkbox onChange={(e) => this.handleSelectAll(e.target.checked)} />
          </th>
          {columns.map(column => (
            <th key={column.id} style={{ width: column.width, textAlign: column.align }}>
              {sortable && column.sortable ? (
                <button 
                  className="sort-button"
                  onClick={() => this.handleSort(column.id)}
                >
                  {column.label}
                  {sortColumn === column.id && (
                    <aava-icon 
                      name={sortDirection === 'asc' ? 'arrow-up' : 'arrow-down'}
                      size="small"
                    />
                  )}
                </button>
              ) : (
                column.label
              )}
            </th>
          ))}
          <th>Actions</th>
        </tr>
      </thead>
    );
  };

  renderTableBody = () => {
    const { data, columns } = this.props;
    const { selectedRows } = this.state;

    return (
      <tbody>
        {data.map(row => (
          <tr key={row.id} className={selectedRows.includes(row.id) ? 'selected' : ''}>
            <td>
              <aava-checkbox 
                checked={selectedRows.includes(row.id)}
                onChange={() => this.handleRowSelect(row.id)}
              />
            </td>
            {columns.map(column => (
              <td key={column.id} style={{ textAlign: column.align }}>
                {this.renderCellContent(row[column.id], column)}
              </td>
            ))}
            <td>
              <div className="row-actions">
                <aava-button size="small" variant="ghost" onClick={() => this.handleEdit(row)}>
                  <aava-icon name="edit" />
                </aava-button>
                <aava-button size="small" variant="ghost" onClick={() => this.handleDelete(row)}>
                  <aava-icon name="trash" />
                </aava-button>
              </div>
            </td>
          </tr>
        ))}
      </tbody>
    );
  };

  renderCellContent = (value: any, column: TableColumn) => {
    if (value === null || value === undefined) {
      return <span className="empty-cell">—</span>;
    }
    
    if (typeof value === 'boolean') {
      return (
        <aava-badge variant={value ? 'success' : 'danger'}>
          {value ? 'Yes' : 'No'}
        </aava-badge>
      );
    }
    
    if (column.id.includes('date') || column.id.includes('time')) {
      return new Date(value).toLocaleDateString();
    }
    
    return String(value);
  };

  render() {
    const { filterable, paginated, pageSize = 10 } = this.props;
    const { isLoading, error, currentPage } = this.state;

    if (error) {
      return (
        <div className="table-error">
          <aava-icon name="alert-circle" />
          <span>Error loading data: {error}</span>
          <aava-button onClick={() => this.setState({ error: null })}>
            Retry
          </aava-button>
        </div>
      );
    }

    return (
      <div className="data-table-container">
        {filterable && (
          <div className="table-filters">
            {/* Filter controls would be rendered here */}
          </div>
        )}
        
        <div className="table-wrapper">
          <table className="data-table">
            {this.renderTableHeader()}
            {this.renderTableBody()}
          </table>
          
          {isLoading && (
            <div className="table-loading">
              <aava-progress-bar indeterminate />
            </div>
          )}
        </div>
        
        {paginated && (
          <div className="table-pagination">
            <aava-pagination
              current-page={currentPage}
              total-pages={Math.ceil(this.props.data.length / pageSize)}
              onPageChange={this.handlePageChange}
            />
          </div>
        )}
      </div>
    );
  }
}
```
