    <ul class="menu-list">

          <aava-icon
            [iconName]="item.icon"
            [iconSize]="20"
            class="menu-icon">
          </aava-icon>
          <span class="menu-label" *ngIf="!collapsed">{{ item.label }}</span>
          <aava-icon
            *ngIf="item.children && !collapsed"
            iconName="chevron-right"
            [iconSize]="16"
            class="expand-icon"
            [class.expanded]="expandedGroups.includes(item.id)">
          </aava-icon>
        </div>
        
        <ul *ngIf="item.children && expandedGroups.includes(item.id) && !collapsed" class="submenu">
          <li *ngFor="let child of item.children" class="submenu-item" [class.active]="isMenuItemActive(child)">
            <div class="submenu-item-content" (click)="onMenuItemClick(child)">
              <span class="submenu-label">{{ child.label }}</span>
            </div>
          </li>
        </ul>
      </li>
    </ul>
  </nav>
  
  <div class="sidebar-footer" *ngIf="showUserContext">
    <div class="user-context" (click)="userProfileClick()">
      <aava-avatars
        size="sm"
        [imageUrl]="userProfile.avatar"
        shape="circle">
      </aava-avatars>
      <div class="user-info" *ngIf="!collapsed">
        <span class="user-name">{{ userProfile.name }}</span>
